public class Car {

// instance members
    private String brand;  
    private String model;   
    private double price;  
    private double mil;
    private String owner;
  
// static members    
private static int totalcars;           
private static String showroomName = "city motors";
      
// constructor 1
   public Car(String brand, String model, double price, double mil) {
      this.brand="kia";     // initialize brand
      this.model="seltos";  // initialize model
      this.price=2000000.d; // initialize price
      this.mil=25.d;    // initialize mileage
      this.owner= "Not Assigned";
      totalcars++;
   }
   
// constructor 2 (overloaded)
   public Car(String brand, String model, double price, double mil, String owner) {
	   this.brand="kia";    
	   this.model="seltos";  
	   this.price=2000000.d;
	   this.mil=25.d;    
	   this.owner= "n";
	   totalcars++;
   }	   

  public String getOwnerName(String owner) {
  return owner;	
   }

public void setOwnerName(String owner) {
   this.owner="owner";
}
  
// instance method
public void displayDetails() {
       System.out.println("Brand: " +brand);
       System.out.println("Model: " +model);
       System.out.println("Price: " +price);
       System.out.println("Mileage: " +mil);
       System.out.println("Owner: " +owner);
   }
public void updatePrice(double newPrice) {
    price = newPrice;
}
public static void showTotalCars() {
    System.out.println("Total Cars: " + totalcars);
}
public static void showShowroomName() {
    System.out.println("Showroom: " + showroomName);
}
}
